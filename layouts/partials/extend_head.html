{{- /* extend_head.html — Inject cross-language hreflang for taxonomy terms whose slugs differ between Chinese and English. Mapping table: data/taxonomy_translations.yaml head.html calls this partial AFTER its own hreflang generation. Uses full language codes (zh-CN / en-US) to match sitemap hreflang values. */ -}} {{- if eq .Kind "term" -}} {{- if lt (len .AllTranslations) 2 -}} {{- $currentLang := .Lang -}} {{- $section := .Section -}} {{- $termSlug := urlize .Data.Term -}} {{- $base := strings.TrimSuffix "/" site.BaseURL -}} {{- $pairs := site.Data.taxonomy_translations.pairs -}} {{- $matched := false -}} {{- range $pairs -}} {{- if eq .section $section -}} {{- $zhSlug := urlize .zh -}} {{- $enSlug := urlize .en -}} {{- $otherLangKey := "" -}} {{- $otherLangCode := "" -}} {{- $otherSlug := "" -}} {{- if and (eq $currentLang "zh") (eq $zhSlug $termSlug) -}} {{- $otherLangKey = "en" -}} {{- $otherLangCode = "en-US" -}} {{- $otherSlug = $enSlug -}} {{- else if and (eq $currentLang "en") (eq $enSlug $termSlug) -}} {{- $otherLangKey = "zh" -}} {{- $otherLangCode = "zh-CN" -}} {{- $otherSlug = $zhSlug -}} {{- end -}} {{- if $otherSlug -}} {{- $matched = true -}} {{- $prefix := cond (eq $otherLangKey "zh") "" (printf "/%s" $otherLangKey) -}} {{- end -}} {{- end -}} {{- end -}} {{- if not $matched -}} {{- warnf "extend_head: term page %q has single hreflang but no mapping found in taxonomy_translations.yaml" .Permalink -}} {{- end -}} {{- end -}} {{- end -}}
